    // After a sentence-terminating period + space, speak should be called

    expect(utterance.text).toBe('Hello world.');
  });

  it('TTS-01: does not speak when disabled', () => {
    const { result } = renderHook(() => useTts({ enabled: false }));
    act(() => { result.current.handleData('Hello world. '); });

  it('TTS-01: partial sentence (no terminator) does not trigger speak', () => {
    const { result } = renderHook(() => useTts({ enabled: true }));
