New App.tsx pattern (locked in RESEARCH.md Pattern 7):
```typescript
const [slopExists, setSlopExists] = useState<boolean | null>(null);
useEffect(() => {
  if (!cwd) { setSlopExists(null); return; }
  fetch(`/api/slop-status?cwd=${encodeURIComponent(cwd)}`)
    .then(r => r.json())
    .then(({ exists }) => setSlopExists(exists))
    .catch(() => setSlopExists(null));
}, [cwd]);
// JSX: {cwd && slopExists === false && <OnboardingModal cwd={cwd} onInit={() => setSlopExists(true)} />}
```
